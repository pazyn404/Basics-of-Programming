#include "iostream"
using namespace std;

int func(int n) {
	if (n == 0) {
		return 1;
	}
	if (n == 1) {
		return 2;
	}
	return pow(func(n - 2), 2) - func(n - 1);
}

int main() {
	int count = 0;
	int sum = 0;
	for (int i = 0; count < 5; i++) {
		if (func(i) % 2 == 0) {
			sum += func(i);
			count++;
		}
	}
	cout << sum;
	cout << '\n';
	system("pause");
	return 0;
}//рекурсія (перевантаження ресурсів)
#include <iostream>
#include <vector>
using namespace std;
int func(int n) {
	static vector<int> vec = { 1,2 };
	if (n != 0 && n != 1) {
		vec.push_back(pow(vec[n - 2], 2) - vec[n - 1]);
	}
	return vec[n];
}

	int count = 0;
	long long sum = 0;

			count++;
		}
	}
	cout << sum;
	cout << '\n';

}//працює стабільно
